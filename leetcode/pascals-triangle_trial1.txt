class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        if numRows==1:
            return [[1]]
        ans = [[1],[1,1]]

        for i in range(numRows-2):
            key = ans[-1]
            temp = [1]

            for i in range(1,len(key)):
                temp.append(key[i]+key[i-1])
            temp.append(1)
            ans.append(temp)
        # print(ans)
        return ans
